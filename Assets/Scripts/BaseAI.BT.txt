tree("Root")
	sequence
		not isStunned
		fallback
			tree("Attack")
			tree("Chase")
			tree("Find")
			tree("Patrol")
		
tree("Find")
	sequence
		WasPlayerSeenLastFrame
		GoToLastKnownPlayerPositionWhileSeekingPlayer
		WaitUnlessPlayerInSight(2.0)

tree("Chase")
	sequence
		HasLineOfSight
		ChasePlayer



tree("Patrol")
	fallback
		sequence
			not HasLineOfSight
			not IsDestinationSet
			FindWaypoint
			SetDestination
		sequence 
			HasArrivedAtDestination
			fallback
				WaitUnlessPlayerInSight(2.0)
				ResetDestination


tree("Attack")
	sequence
		IsNearPlayer
		while not isStunned
			sequence
				WindupAttack
				LookAtPlayerWhileWindingUp(.7)
				Attack
				Wait(2.0)
