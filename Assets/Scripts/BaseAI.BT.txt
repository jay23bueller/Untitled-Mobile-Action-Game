tree("Root")
	sequence
		not isStunned
		race
			tree("Attack")
			tree("Chase")
			tree("Patrol")
		
		

tree("Chase")
	sequence
		HasLineOfSight
		ChasePlayer


tree("Patrol")
	fallback
		sequence
			not HasLineOfSight
			not IsDestinationSet
			FindWaypoint
			SetDestination
		sequence 
			HasArrivedAtDestination
			fallback
				WaitUnlessPlayerInSight(2.0)
				ResetDestination


tree("Attack")
	sequence
		IsNearPlayer
		WindupAttack
		Wait(0.7)
		Attack
		Wait(2.0)
